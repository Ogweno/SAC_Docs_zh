\SACCMD{border}
\label{cmd:border}

\SACTitle{概要}
控制图形四周边框的绘制

\SACTitle{语法}
\begin{SACSTX}
BORDER [ON|OFF]
\end{SACSTX}

\SACTitle{输入}
\begin{description}
\item [ON|OFF] 绘制/不绘制图形四周的边框
\end{description}

\SACTitle{缺省值}
\begin{SACDFT}
border off
\end{SACDFT}

\SACTitle{说明}
参考 ``\nameref{sec:plot-appearance}'' 一节。
